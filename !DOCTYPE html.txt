        // --- RENDER LOGIC ---

        function goToPackageCatalog() {
            switchPage('catalog');
            filterCatalog('packages');
            window.scrollTo({top: 0, behavior: 'instant'}); 
        }

        function renderFilters() {
            const container = document.getElementById('category-filters');
            let html = '<button onclick="filterCatalog(\'all\')" class="filter-btn active bg-brand-black text-white px-4 py-2 border-2 border-brand-black font-bold uppercase text-sm shadow-hard transform scale-105" data-cat="all">Semua</button>';
            
            CATEGORIES.forEach(cat => {
                html += `<button onclick="filterCatalog('${cat.id}')" class="filter-btn bg-white text-brand-black px-4 py-2 border-2 border-brand-black font-bold uppercase text-sm hover:bg-brand-gray hover:shadow-hard transition-all" data-cat="${cat.id}">${cat.name}</button>`;
            });
            container.innerHTML = html;
        }

        function filterCatalog(category) {
            // Update Button Styles (Brutalist toggle)
            document.querySelectorAll('.filter-btn').forEach(btn => {
                if(btn.dataset.cat === category) {
                    btn.className = "filter-btn active bg-brand-black text-white px-4 py-2 border-2 border-brand-black font-bold uppercase text-sm shadow-hard transform scale-105";
                } else {
                    btn.className = "filter-btn bg-white text-brand-black px-4 py-2 border-2 border-brand-black font-bold uppercase text-sm hover:bg-brand-gray hover:shadow-hard transition-all";
                }
            });

            const items = category === 'all' ? PRODUCTS : PRODUCTS.filter(p => p.category === category);
            const container = document.getElementById('catalog-grid');
            let html = '';
            
            if (items.length === 0) {
                 html = `<div class="col-span-full text-center py-20 text-gray-400 font-bold uppercase">Belum ada produk.</div>`;
            } else {
                items.forEach(item => {
                    html += `
                    <div class="group border-2 border-brand-black bg-white hover:shadow-hard transition-all duration-200">
                        <div class="aspect-[4/3] bg-gray-100 relative overflow-hidden border-b-2 border-brand-black">
                            <img src="./images/${item.img}" onerror="this.style.display='none'; this.nextElementSibling.style.display='flex'" class="w-full h-full object-cover grayscale group-hover:grayscale-0 transition-all duration-500">
                            <!-- Fallback -->
                            <div class="hidden flex-col items-center justify-center absolute inset-0 text-gray-400">
                                <i data-lucide="image" class="w-12 h-12 mb-2 opacity-30"></i>
                            </div>
                        </div>
                        <div class="p-4">
                            <div class="text-[10px] font-bold text-white bg-brand-black px-2 py-0.5 w-fit mb-2 uppercase">${item.category === 'packages' ? 'PAKET' : item.category}</div>
                            <h3 class="font-heading font-bold text-xl leading-tight mb-2 uppercase">${item.name}</h3>
                            <p class="text-xs text-gray-600 font-medium mb-4 leading-relaxed border-l-2 border-brand-orange pl-2">${item.desc}</p>
                            <a href="https://wa.me/6281234567890?text=Tanya%20harga%20${item.name}" target="_blank" class="block w-full text-center py-2 border-2 border-brand-black font-bold uppercase text-sm hover:bg-brand-black hover:text-white transition-colors">
                                Minta Harga
                            </a>
                        </div>
                    </div>`;
                });
            }
            container.innerHTML = html;
            lucide.createIcons();
        }

        // --- NAVIGATION ---
        function switchPage(pageId) {
            document.querySelectorAll('.page-section').forEach(el => el.classList.add('hidden'));
            document.getElementById(pageId + '-page').classList.remove('hidden');
            document.getElementById('mobile-menu').classList.add('hidden');
            window.scrollTo({top: 0, behavior: 'instant'});
        }

        function toggleMobileMenu() {
            document.getElementById('mobile-menu').classList.toggle('hidden');
        }

        // Init
        window.addEventListener('DOMContentLoaded', () => {
            lucide.createIcons();
            renderFilters();
            filterCatalog('all');
        });
